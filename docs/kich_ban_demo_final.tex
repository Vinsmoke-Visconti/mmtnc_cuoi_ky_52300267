\section{PHẦN 1: KHỞI ĐỘNG VÀ ĐỐI CHIẾU KIẾN TRÚC MẠNG (MỨC 2 \& MỨC 3)}

\subsection{1. Khởi động hệ thống}
\textbf{Thao tác:} Mở Terminal tại thư mục \texttt{mmtnc\_cuoi\_ky\_52300267} và chạy lệnh:
\begin{lstlisting}[language=bash]
sudo python3 main_topology.py
\end{lstlisting}
Đợi hệ thống khởi tạo các node, gán IP, cấu hình NAT, và chờ 30 giây để OSPF hội tụ. Màn hình sẽ hiện ra dấu nhắc \texttt{mininet>}.

\subsection{2. Trình bày sự khớp nối với đề bài}
\textbf{Lời thoại trình bày:}
"Thưa thầy/cô, hệ thống của em được thiết kế chuẩn theo mô hình \textbf{Campus 3 lớp} có tích hợp DMZ, đáp ứng toàn bộ yêu cầu của đề bài:
\begin{itemize}
    \item \textbf{Lớp Access:} Gồm 2 switch \texttt{s2}, \texttt{s3} kết nối người dùng thuộc 2 VLAN 192.168.10.0/24 và 192.168.20.0/24.
    \item \textbf{Lớp Distribution:} Gồm \texttt{dist1} và \texttt{dist2} làm Default Gateway. \textit{(Điểm Mức 3: Em đặt Layer 3 tại đây để giới hạn Broadcast Domain, giảm tải cho switch Access).}
    \item \textbf{Lớp Core:} Router \texttt{core} đảm nhận chuyển mạch tốc độ cao.
    \item \textbf{Vùng DMZ:} Tách biệt vật lý thông qua \texttt{s1}, chứa Web1 và Web2.
    \item \textbf{Biên mạng (Edge):} Router \texttt{fw} thực hiện NAT, ACL và giám sát bảo mật.
    \item \textbf{Tính sẵn sàng cao (HA - Mức 3):} Em thiết kế thêm một đường \textbf{HA Link} nối trực tiếp giữa \texttt{dist1} và \texttt{dist2} để dự phòng sự cố."
\end{itemize}

\section{PHẦN 2: KIỂM TRA KẾT NỐI VÀ PHÂN TÍCH LUỒNG GÓI TIN (MỨC 2)}

\subsection{1. Ping liên mạng nội bộ (Inside to Inside)}
\textbf{Thao tác (Mininet CLI):}
\begin{lstlisting}[language=bash]
mininet> h1 ping -c 1 h3
\end{lstlisting}
\textbf{Lời thoại trình bày luồng gói tin:}
"Khi h1 (192.168.10.11) ping h3 (192.168.20.11):
1. Gói tin đi từ h1 lên Default Gateway là \texttt{dist1}.
2. \texttt{dist1} tra bảng định tuyến OSPF, đẩy lên \texttt{core}.
3. \texttt{core} nhận thấy mạng đích nằm ở nhánh \texttt{dist2}, đẩy gói tin xuống \texttt{dist2}.
4. \texttt{dist2} đẩy xuống switch Access 2 và tới h3. Gói tin phản hồi đi ngược lại tương tự."

\subsection{2. Ping từ Inside ra Internet (Kiểm chứng PAT)}
\textbf{Thao tác:}
\begin{lstlisting}[language=bash]
mininet> h1 ping -c 1 ext
\end{lstlisting}
\textbf{Lời thoại trình bày luồng gói tin:}
"Khi h1 ping ext (203.0.113.1):
1. Gói tin lên \texttt{dist1} $\rightarrow$ \texttt{core} $\rightarrow$ \texttt{fw}.
2. Tại \texttt{fw}, do chính sách \textbf{PAT (Overload)}, địa chỉ nguồn 192.168.10.11 bị \texttt{iptables} dịch (MASQUERADE) thành IP Public của Firewall là 203.0.113.2 kèm theo một Source Port ngẫu nhiên.
3. Gói tin đến \texttt{ext}. Khi \texttt{ext} trả lời về 203.0.113.2, \texttt{fw} tra bảng NAT Session, dịch ngược đích về 192.168.10.11 và trả lại cho h1."

\subsection{3. Truy cập từ Internet vào DMZ (Kiểm chứng Static NAT)}
\textbf{Thao tác:}
\begin{lstlisting}[language=bash]
mininet> ext curl -s 100.0.0.11:80
\end{lstlisting}
\textbf{Lời thoại trình bày luồng gói tin:}
"Khi ext truy cập IP Public ảo 100.0.0.11:
1. Gói tin đến cổng ngoài của \texttt{fw}.
2. Quá trình \textbf{DNAT (Destination NAT)} xảy ra, đổi IP đích từ 100.0.0.11 thành IP thật của Web1 là 10.10.10.11.
3. FW kiểm tra ACL, thấy Port 80 hợp lệ, cho phép gói tin vào vùng DMZ.
4. Khi Web1 trả lời, quá trình \textbf{SNAT (Source NAT)} dịch nguồn từ 10.10.10.11 ngược lại thành 100.0.0.11 để trả về cho ext."

\section{PHẦN 3: ĐỊNH TUYẾN OSPF VÀ TÍNH SẴN SÀNG CAO HA (MỨC 3)}

\subsection{1. Kiểm tra trạng thái hội tụ OSPF}
\textbf{Thao tác:}
\begin{lstlisting}[language=bash]
mininet> fw vtysh --vty_socket /tmp/frr_fw -c "show ip ospf neighbor"
mininet> core vtysh --vty_socket /tmp/frr_core -c "show ip route"
\end{lstlisting}
\textbf{Lời thoại:} "Tất cả các Router đều đạt trạng thái \textbf{Full}, chứng tỏ OSPF Area 0 đã hội tụ hoàn toàn. Bảng định tuyến của Core hiện đầy đủ các tuyến đường chữ 'O' (học từ OSPF)."

\subsection{2. Demo cơ chế dự phòng HA (Failover)}
\textbf{Thao tác:} (Mở thêm một Terminal mới, KHÔNG tắt Mininet)
\begin{lstlisting}[language=bash]
cd mmtnc_cuoi_ky_52300267
sudo python3 scripts/test_ha.py
\end{lstlisting}
\textbf{Lời thoại:}
"Để chứng minh Mức 3, em tiến hành đánh sập đường link chính giữa \texttt{dist1} và \texttt{core} trong lúc h1 đang ping liên tục. 
Hệ thống mất khoảng \textbf{3.4 giây} để hội tụ lại. OSPF đã tự động phát hiện sự cố và định tuyến luồng dữ liệu chạy vòng qua đường \textbf{HA Link (dist1 $\rightarrow$ dist2 $\rightarrow$ core)}. Sau sự cố, kết nối hoàn toàn khôi phục mà không cần can thiệp thủ công."

\section{PHẦN 4: BẢO MẬT ĐA LỚP VÀ TRỰC QUAN HÓA (MỨC 2 \& 3)}

\subsection{1. Kiểm chứng Extended ACL và Standard ACL (Mức 2)}
\textbf{Thao tác:}
\begin{lstlisting}[language=bash]
# 1. Kiem tra Inside vao DMZ (Duoc phep HTTP)
mininet> h1 curl -s 10.10.10.11:80
# 2. Kiem tra VLAN 20 (h3) SSH vao DMZ (Bi Drop boi Standard ACL)
mininet> h3 ssh 10.10.10.11  (Nhan Ctrl+C de ngat neu bi treo)
# 3. Kiem tra DMZ vao Inside (Bi block 1 chieu)
mininet> web1 ping -c 1 192.168.10.11
\end{lstlisting}
\textbf{Lời thoại (Logic Mức 3):}
"Hệ thống thiết lập Zero-Trust (Default DROP). 
- \textbf{Extended ACL} (chỉ cho phép port 80/443) được em đặt ngay tại đầu vào của mạng nội bộ để loại bỏ gói tin rác sớm nhất, tiết kiệm băng thông. 
- \textbf{Standard ACL} chặn h3 SSH vào Web1 hoạt động chính xác.
- Server DMZ hoàn toàn không thể chủ động kết nối ngược vào Inside, chống Pivot Attack."

\subsection{2. Phân tích Truy vết NAT và Heatmap (Mức 3)}
\textbf{Thao tác:}
Mở file ảnh \texttt{results/heatmap\_acl.png} (nếu đã chạy \texttt{sudo python3 scripts/security\_heatmap.py}).
\textbf{Lời thoại:}
"Thưa thầy/cô, việc dùng PAT có nhược điểm là làm mất \textbf{Traceability (tính truy vết)}. Để khắc phục (Mức 3), em đã bật tính năng Logging trên Iptables. Khi có sự cố, ta có thể đối chiếu Source Port từ log ISP với bảng Log Syslog của Linux để tìm ra chính xác IP nội bộ nào đã gây họa.
Ngoài ra, log của các gói tin bị ACL Drop cũng được thu thập để vẽ thành \textbf{Security Heatmap}. Dựa vào các mảng màu đỏ, quản trị viên có thể ngay lập tức nhận diện địa chỉ IP nào đang cố gắng dò quét tấn công mạng vào khung giờ nào."

\section{PHẦN 5: CÂN BẰNG TẢI DỰA TRÊN NGƯỠNG (MỨC 2 \& 3)}

\subsection{1. Demo thuật toán Load Balancing}
\textbf{Thao tác (trong Mininet CLI):}
\begin{lstlisting}[language=bash]
mininet> test_lb
\end{lstlisting}
\textbf{Lời thoại trình bày:}
"Kịch bản Load Balancing của em hoạt động tự động thông qua Python giám sát băng thông:
1. Script sẽ tiêm một luồng traffic lớn (iPerf) vào IP Public 100.0.0.11.
2. Khi băng thông Web1 (Primary) vượt \textbf{ngưỡng 80\%}, Python ngay lập tức xóa rule DNAT cũ và thêm rule DNAT mới chuyển hướng các kết nối mới sang Web2 (Backup). Trên màn hình CLI sẽ báo \texttt{>> CHUYEN -> web2}.
3. Khi tải trên Web1 giảm xuống \textbf{dưới 20\%}, hệ thống nhận diện Web1 đã an toàn, tự động cấu hình iptables trả lại luồng traffic về Web1."

\subsection{2. Đồ thị và Phân tích hiệu năng}
\textbf{Thao tác:}
Mở file Excel báo cáo \texttt{results/mmtnc\_report\_*.xlsx} (hoặc các hình ảnh biểu đồ trong folder results).
\textbf{Lời thoại:}
"Để minh chứng, script của em sử dụng \textbf{Matplotlib} để vẽ đồ thị đường (Line Chart). Các đường đứt nét trên đồ thị minh họa chính xác thời điểm tải chạm ngưỡng 80\% và sự gia tăng băng thông tương ứng trên Server dự phòng.

Đồng thời, bảng thống kê hiệu năng cho thấy: Khi gói tin phải đi qua Firewall xử lý NAT và kiểm tra ACL, \textbf{độ trễ (Latency) bị tăng thêm khoảng 1.3ms} (NAT Overhead - Tiêu chí Mức 3). Việc tốn thêm RAM/CPU để duy trì bảng \texttt{conntrack} là sự đánh đổi bắt buộc để có được sự an toàn cho toàn hệ thống."
